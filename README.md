sqwiggle-node
=============

Node.js library for the Sqwiggle API

### Install

    npm install sqwiggle-node

### Usage

    var Sqwiggle = require "sqwiggle-node";

    var token = '...'; // Your API token here

    var app = new Sqwiggle(token);
